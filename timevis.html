4 timevis

4.1 类型

library(timevis)
timevis(
  data.frame(id = 1:4,
             content = c("one", "two","three", "four"),
             start = c("2016-01-10", "2016-01-12","2016-01-14","2016-01-16"),
            end = c("2016-01-18", "2016-01-20","2016-01-22","2016-01-24"),
             type=c("box","point","range","background")
))
